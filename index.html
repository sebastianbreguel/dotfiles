Quick Start

# 1. Install Homebrew
/bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"

# 2. Clone this repo
git clone https://github.com/sebastianbreguel/dotfiles.git ~/dotfiles

# 3. Copy Claude Code config
cp -r ~/dotfiles/.claude/ ~/.claude/

# 4. Run the full setup script (see README.md)
cat ~/dotfiles/README.md
